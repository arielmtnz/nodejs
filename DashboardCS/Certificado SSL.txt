Generar el archivo .cnf para los certificados en la carpeta certs de nginx.
openssl version

openssl req -x509 -nodes -days 365 -newkey rsa:2048  -keyout C:/nginx/certs/dashboardcenturysystems.key   -out C:/nginx/certs/dashboardcenturysystems.crt   -subj "/CN=dashboardcenturysystems"   -addext "subjectAltName = DNS:dashboardcenturysystems"
